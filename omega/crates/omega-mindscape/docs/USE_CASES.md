## Getting Started

```rust
use omega_mindscape::MindscapeExplorer;

// Create a new mindscape
let mut explorer = MindscapeExplorer::new(1000.0);

// Store memories with embeddings
explorer.remember("memory_label".to_string(), embedding_vector);

// Navigate the mindscape
let nearby = explorer.look_around(5);

// Enter dream state for discovery
explorer.enter_dream_state();
if let Some(vision) = explorer.dream() {
    println!("Dream vision: {:?}", vision);
}

// Observe yourself observing (strange loops)
let observation = explorer.observe_self(5);
if observation.detected_strange_loop {
    println!("Strange loop detected at depth {}", observation.depth);
}
```
