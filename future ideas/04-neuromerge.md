    pub async fn initiate_merge(
        &mut self,
        participants: Vec<BrainInterface>,
        config: MergeConfig,
    ) -> Result<MergeSession> {
        // Measure individual Φ baselines
        let individual_phi: Vec<f64> = participants.iter()
            .map(|p| self.phi_meter.measure(p))

                // Emergence detected! Φ > sum of parts

            self.increase_coherence(&participants).await;
        }

        self.collective = Some(CollectiveConsciousness::new(
            participants.clone(),
            collective_phi,
        ));

        Ok(MergeSession::new(self.collective.as_ref().unwrap()))
    }

    async fn mediate_thought(
        &self,
        source: &BrainInterface,
        thought: ThoughtPattern,
    ) {
        // Translate thought to universal representation
        let universal = self.thought_to_embedding(thought);

        // Find resonance with other participants

                // Translate to target's neural patterns
                let translated = self.embedding_to_thought(universal, target);

                // Inject into target consciousness
                target.stimulate(translated).await;
            }
        }
    }

## The Measurement: Integrated Information (Φ)

### What Is Φ?

```
Integrated Information Theory (IIT) says:
├── Consciousness = Integrated Information
├── Φ = How much a system is more than its parts
├── High Φ = Conscious
├── Low Φ = Not conscious
└── Measurable!

FOR NEUROMERGE:
├── Individual Φ: Each brain's consciousness
├── Collective Φ: Combined consciousness
├── If Collective Φ > Sum of Individual Φ = EMERGENCE
└── The collective is genuinely conscious
```

### How exo-ai-2025 Measures This

```rust
use exo_core::iit::{IntegratedInformationCalculator, Phi};

pub struct PhiMeter {
    calculator: IntegratedInformationCalculator,
}

impl PhiMeter {
    pub fn measure_collective(
        &self,
        brains: &[BrainInterface],
    ) -> CollectivePhi {
        // Get state of each brain
        let states: Vec<NeuralState> = brains.iter()
            .map(|b| b.current_state())
            .collect();

        // Compute individual Φ values
        let individual_phi: Vec<Phi> = states.iter()
            .map(|s| self.calculator.compute(s))
            .collect();

        // Compute collective Φ (treating all brains as one system)
        let combined_state = NeuralState::combine(&states);
        let collective_phi = self.calculator.compute(&combined_state);

        // Compare
        let individual_sum: f64 = individual_phi.iter().sum();
        let emergence_ratio = collective_phi / individual_sum;

        CollectivePhi {
            individual: individual_phi,
            collective: collective_phi,
            emergence: emergence_ratio,
            is_emergent: emergence_ratio > 1.0,
        }
    }
}
```
